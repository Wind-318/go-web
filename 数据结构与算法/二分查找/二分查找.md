## 二分查找
二分查找是一种对**有序序列**进行查找的算法。
- 寻找两端都闭的区间
    ```C++
    int search(const vector<int>& arr, int target) {
        int left = 0;
        int right = arr.size();
        while (left + 1 < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] == target) {
                return mid;
            } else if (arr[mid] <= target) {
                left = mid;
            } else {
                right = mid;
            }
        }

        return -1;
    }
    ```
- 寻找左侧边界、左闭右开的区间
    ```C++
    int search(const vector<int>& arr, int target) {
        int left = 0;
        int right = arr.size();
        if (right == 0) {
            return -1;
        }

        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] == target) {
                right = mid;
            } else if (arr[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        if (left == arr.size()) {
            return -1;
        }
        return arr[left] == target ? left : -1;
    }
    ```
- 寻找右侧边界、左开右闭的区间
    ```C++
    int search(const vector<int>& arr, int target) {
        int left = 0;
        int right = arr.size();
        if (right == 0) {
            return -1;
        }

        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] == target) {
                left = mid + 1;
            } else if (arr[mid] < target) {
                left = mid + 1;
            } else {
                right = mid;
            }
        }

        if (left == 0) {
            return -1;
        }
        return arr[left-1] == target ? (left-1) : -1;
    }
    ```
